function toggleFilterPanel() {
    const panel   = document.getElementById('filter-panel');
    const overlay = document.getElementById('filter-overlay');
    const btn     = document.getElementById('filter-panel-btn');
    const isOpen  = !panel.classList.contains('translate-x-full');
    if (isOpen) {
        // Close
        panel.classList.add('translate-x-full');
        overlay.classList.add('hidden');
        btn.classList.remove('bg-orange-500','text-white','border-orange-500');
        btn.classList.add('bg-orange-50','text-orange-700','border-orange-200');
    } else {
        // Open
        panel.classList.remove('translate-x-full');
        overlay.classList.remove('hidden');
        btn.classList.add('bg-orange-500','text-white','border-orange-500');
        btn.classList.remove('bg-orange-50','text-orange-700','border-orange-200');
        populateFilterDropdowns();
    }
}

function populateFilterDropdowns() {
    // Sources
    const sources = [...new Set(leadsData.map(l => l.source).filter(s => s && s !== '-'))].sort();
    const srcSel  = document.getElementById('fp-source');
    const curSrc  = srcSel.value;
    srcSel.innerHTML = '<option value="">All Sources</option>' + sources.map(s => `<option${s===curSrc?' selected':''}>${s}</option>`).join('');
    // Users
    const users   = [...new Set(leadsData.map(l => l.user).filter(u => u && u !== '-'))].sort();
    const usrSel  = document.getElementById('fp-user');
    const curUsr  = usrSel.value;
    usrSel.innerHTML = '<option value="">All Users</option>' + users.map(u => `<option${u===curUsr?' selected':''}>${u}</option>`).join('');
}

function downloadFilteredExcel() {
    // Get current filtered leads (re-run filter logic inline)
    let processed = [...leadsData];
    if (advFilters.name)    processed = processed.filter(l => (l.name||'').toLowerCase().includes(advFilters.name));
    if (advFilters.mobile)  processed = processed.filter(l => (l.mobile||'').toLowerCase().includes(advFilters.mobile));
    if (advFilters.email)   processed = processed.filter(l => (l.email||'').toLowerCase().includes(advFilters.email));
    if (advFilters.stage)   processed = processed.filter(l => (l.stage||'').toUpperCase() === advFilters.stage.toUpperCase());
    if (advFilters.source)  processed = processed.filter(l => (l.source||'') === advFilters.source);
    if (advFilters.user)    processed = processed.filter(l => (l.user||'') === advFilters.user);
    if (advFilters.priority) processed = processed.filter(l => (l.priority||'').toLowerCase() === advFilters.priority.toLowerCase());
    if (advFilters.assign === 'assigned')   processed = processed.filter(l => l.user && l.user !== '-' && l.user !== '');
    if (advFilters.assign === 'unassigned') processed = processed.filter(l => !l.user || l.user === '-' || l.user === '');
    if (advFilters.dateFrom) { const from = new Date(advFilters.dateFrom).getTime(); processed = processed.filter(l => l.date >= from); }
    if (advFilters.dateTo)   { const to   = new Date(advFilters.dateTo).getTime() + 86399999; processed = processed.filter(l => l.date <= to); }

    // Build CSV
    const headers = ['Name','Mobile','Email','Stage','Source','User','Priority','Creation Date','Modified Date'];
    const rows = processed.map(l => [
        `"${(l.name||'').replace(/"/g,'""')}"`,
        `"${(l.mobile||'')}"`,
        `"${(l.email||'')}"`,
        `"${(l.stage||'')}"`,
        `"${(l.source||'')}"`,
        `"${(l.user||'')}"`,
        `"${(l.priority||'')}"`,
        `"${(l.creationDate||'')}"`,
        `"${(l.modifiedDate||'')}"`
    ].join(','));
    const csv = [headers.join(','), ...rows].join('\n');
    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url; a.download = 'leads_filtered_' + new Date().toISOString().slice(0,10) + '.csv';
    document.body.appendChild(a); a.click();
    document.body.removeChild(a); URL.revokeObjectURL(url);
}

function updateFilterChips() {
    const chips = document.getElementById('filter-chips');
    const badge = document.getElementById('filter-active-badge');
    if (!chips) return;
    const labels = {
        name:'Name', mobile:'Mobile', email:'Email', assign:'Assign',
        userGroup:'Group', user:'User', stage:'Stage', source:'Source',
        priority:'Priority', adsName:'Ads', country:'Country', city:'City',
        pincode:'Pincode', tag:'Tag', segment:'Segment',
        dateFrom:'From', dateTo:'To', customFieldName:'Field', customFieldValue:'Value'
    };
    const active = Object.entries(advFilters).filter(([k,v]) => v && k !== 'dateType');
    badge.textContent = active.length;
    badge.classList.toggle('hidden', active.length === 0);
    chips.innerHTML = active.map(([k,v]) =>
        `<span class="inline-flex items-center gap-1 px-2.5 py-1 bg-orange-100 border border-orange-200 text-orange-700 text-[11px] font-bold rounded-full">
            ${labels[k]||k}: ${v}
            <button onclick="clearOneFilter('${k}')" class="material-symbols-outlined text-[12px] hover:text-orange-900 cursor-pointer leading-none">close</button>
        </span>`
    ).join('');
}

    if (el) el.value = '';
    currentPage = 1;
    renderLeads();
    updateFilterChips();
}

function updateSummaryCards() {
    let filteredForCounts = [...leadsData];
    if (searchQuery) {
        filteredForCounts = filteredForCounts.filter(l =>
            l.name.toLowerCase().includes(searchQuery) ||
            l.email.toLowerCase().includes(searchQuery) ||
            l.mobile.toLowerCase().includes(searchQuery) ||
            l.user.toLowerCase().includes(searchQuery) ||
            l.source.toLowerCase().includes(searchQuery) ||
            (l.stage && l.stage.toLowerCase().includes(searchQuery))
        );
    }

    const counts = {
        "ALL": filteredForCounts.length,
        "LEAD": 0,
        "LOST": 0,
        "FOLLOW UP": 0,
        "PROPOSAL SENT": 0,
        "OTHERS": 0,
        "HOT": 0,
        "TRIAL AC": 0,
        "ONBOARDED": 0,
        "DEMO DONE": 0,
        "RNR": 0,
        "CALL BACK": 0,
        "INVOICE SENT": 0,
        "DEMO SCHEDULE": 0,
        "DUPLICATE": 0,
        "DIRECT MEETING": 0
    };

    filteredForCounts.forEach(l => {
        const rawStage = (l.stage || '').toUpperCase().trim();
        let matched = false;
        for (const cardKey of Object.keys(counts)) {
            if (cardKey === "ALL") continue;
            const aliases = STATUS_ALIASES[cardKey] || [cardKey];
            if (aliases.includes(rawStage)) {
                counts[cardKey]++;
                matched = true;
                break;
            }
        }
        if (!matched && rawStage) {
            const found = Object.keys(counts).find(k => k === rawStage);

                counts[found]++;
            } else {
                counts["OTHERS"]++;

                    const name = (lead.name || '').trim().toLowerCase();
                    const mobile = (lead.mobile || '').trim();
                    
                    if (!name) return;
                    if (seenNames.has(name)) return;
                    if (mobile && mobile !== '-' && seenMobiles.has(mobile)) return;
                    
                    seenNames.add(name);
                    if (mobile && mobile !== '-') seenMobiles.add(mobile);
                    
                    // Clear old hardcoded segment values as per user request
                    if (lead.segment === "Enterprise" || lead.segment === "Mid-Market" || lead.segment === "SMB") {
                        lead.segment = "";
                    }
                    leadsData.push(lead);
                });
                saveLeadsData();
                return;
            }
        }
    } catch(e) {}
    leadsData = [];
    saveLeadsData();
}

    currentPage += direction;
    renderLeads();
    const mainEl = document.querySelector('main');
    if (mainEl) mainEl.scrollTop = 0;

function changePageSize(size) {
    pageSize = size === 'All' ? 'All' : parseInt(size);
    currentPage = 1;
    renderLeads();
    const mainEl = document.querySelector('main');
    if (mainEl) mainEl.scrollTop = 0;

function updatePaginationControls(totalPages) {
    const prevBtn = document.getElementById('btn-prev');
    const nextBtn = document.getElementById('btn-next');
    if (prevBtn) prevBtn.disabled = currentPage === 1;
    if (nextBtn) nextBtn.disabled = currentPage === totalPages;

    const pageNumbersDiv = document.getElementById('page-numbers');
    if (pageNumbersDiv) {
        pageNumbersDiv.innerHTML = '';

        const startRange = Math.max(1, currentPage - 2);
        const endRange = Math.min(totalPages, currentPage + 2);

        for (let i = startRange; i <= endRange; i++) {
            const btn = document.createElement('button');
            btn.textContent = i;
            if (i === currentPage) {
                btn.className = 'w-8 h-8 flex items-center justify-center rounded-xl bg-[#0D9488] text-white text-xs font-bold shadow-sm cursor-default';
            } else {
                btn.className = 'w-8 h-8 flex items-center justify-center rounded-xl bg-white border border-slate-200 text-slate-600 hover:bg-slate-100 text-xs font-bold cursor-pointer transition-colors';
                btn.onclick = () => {
                    currentPage = i;
                    renderLeads();
                    const mainEl = document.querySelector('main');
                    if (mainEl) mainEl.scrollTop = 0;
                };
            }
            pageNumbersDiv.appendChild(btn);
        }

function toggleTableColumn(checkbox) {
    if (checkbox.checked) hiddenColumns.delete(checkbox.value);
    else hiddenColumns.add(checkbox.value);
    saveHiddenColumns();
    applyColumnVisibility();
    updateCustomizeLabel();
}

function applyColumnVisibility() {
    document.querySelectorAll('th, td').forEach(cell => {
        const col = Array.from(cell.classList).find(c => c.startsWith('col-'));
        if (col) cell.classList.toggle('hidden', hiddenColumns.has(col));
    });
    const emptyCell = document.querySelector('#leads-table-body td[colspan]');
    if (emptyCell && emptyCell.textContent.trim() === 'No records found') {
        emptyCell.setAttribute('colspan', 18 - hiddenColumns.size);
    }
}

function updateCustomizeLabel() {
    document.getElementById('customize-label').textContent = `${15 - hiddenColumns.size} items selected`;
}

function saveHiddenColumns() {
    try { localStorage.setItem('hiddenColumns', JSON.stringify(Array.from(hiddenColumns))); } catch(e) {}
}

function loadHiddenColumns() {
    try {
        const stored = localStorage.getItem('hiddenColumns');
        if (stored) {
            const arr = JSON.parse(stored);
            if (Array.isArray(arr)) {
                hiddenColumns.clear();
                arr.forEach(c => hiddenColumns.add(c));
            }
        }
    } catch(e) {}
    
    document.querySelectorAll('#customize-dropdown input[type="checkbox"]').forEach(cb => {
        cb.checked = !hiddenColumns.has(cb.value);
    });
    updateCustomizeLabel();
}

// ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ Bulk Action ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬ГўвЂќв‚¬
function handleBulkAction(action) {
    if (!action) return;
    
    if (action === 'delete-all') {
        if (confirm('Are you sure you want to delete ALL leads in the database? This action cannot be undone.')) {
            leadsData = [];
            saveLeadsData();
            renderLeads();
        }
        document.getElementById('bulk-action-select').value = '';
        return;
    }

    const selected = Array.from(document.querySelectorAll('.lead-checkbox')).filter(cb => cb.checked).map(cb => cb.value);
    if (selected.length === 0) { 
        alert('Please select at least one lead first.'); 
        document.getElementById('bulk-action-select').value = ''; 
        return; 
    }

    if (action === 'assign-user') {
        const targetUser = prompt(`Enter username to assign to the selected ${selected.length} lead(s):`);
        if (targetUser && targetUser.trim()) {
            leadsData.forEach(l => {
                if (selected.includes(l.name)) {
                    l.user = targetUser.trim();
                    l.modifiedDate = new Date().toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });
                }
            });
            saveLeadsData();
            renderLeads();
        }
    } else if (action === 'change-status') {
        const targetStatus = prompt(`Enter new status/stage for the selected ${selected.length} lead(s) (e.g. LEAD, FOLLOW UP, ONBOARDED):`);
        if (targetStatus && targetStatus.trim()) {
            const formattedStatus = targetStatus.trim().toUpperCase();
            leadsData.forEach(l => {
                if (selected.includes(l.name)) {
                    l.stage = formattedStatus;
                    l.status = formattedStatus;
                    l.modifiedDate = new Date().toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });
                }
            });
            saveLeadsData();
            renderLeads();
        }
    } else if (action === 'change-followup') {
        const targetDate = prompt(`Enter new follow-up date (YYYY-MM-DD) for the selected ${selected.length} lead(s):`);
        if (targetDate && targetDate.trim()) {
            leadsData.forEach(l => {
                if (selected.includes(l.name)) {
                    l.followupDate = targetDate.trim();
                    l.modifiedDate = new Date().toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });
                }
            });
            saveLeadsData();
            renderLeads();
        }
    } else if (action === 'create-autodialer') {
        if (confirm(`Create an autodialer campaign for the selected ${selected.length} lead(s)?`)) {
            alert(`Autodialer campaign successfully created for ${selected.length} lead(s).`);
        }
    } else if (action === 'send-sms') {
        const msg = prompt(`Enter SMS message content to send to ${selected.length} lead(s):`);
        if (msg && msg.trim()) {
            alert(`SMS campaign scheduled successfully for ${selected.length} lead(s).`);
        }
    } else if (action === 'send-email') {
        const subject = prompt('Enter Email subject:');
        if (subject && subject.trim()) {
            const body = prompt('Enter Email body:');
            if (body && body.trim()) {
                alert(`Email campaign scheduled successfully for ${selected.length} lead(s).`);
            }
        }
    } else if (action === 'send-whatsapp') {
        const msg = prompt(`Enter WhatsApp message to send to ${selected.length} lead(s):`);
        if (msg && msg.trim()) {
            alert(`WhatsApp messages scheduled successfully for ${selected.length} lead(s).`);
        }
    } else if (action === 'create-segment') {
        const segmentName = prompt(`Enter new segment name for the selected ${selected.length} lead(s):`);
        if (segmentName && segmentName.trim()) {
            leadsData.forEach(l => {
                if (selected.includes(l.name)) {
                    l.segment = segmentName.trim();
                    l.modifiedDate = new Date().toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });
                }
            });
            saveLeadsData();
            renderLeads();
        }
    }
    
    document.getElementById('bulk-action-select').value = '';
}

function switchAddLeadTab(tabIndex) {
    if (tabIndex < 0 || tabIndex > 2) return;
    currentAddLeadTab = tabIndex;
    document.querySelectorAll('.add-lead-tab').forEach((btn, i) => {
        if (i === tabIndex) {
            let activeStyle = '';
            if (i === 0) activeStyle = 'bg-[#0D9488] hover:bg-[#0f766e] text-white shadow-md font-bold';
            else if (i === 1) activeStyle = 'bg-[#0ea5e9] hover:bg-[#0284c7] text-white shadow-md font-bold';
            else if (i === 2) activeStyle = 'bg-[#D97706] hover:bg-[#b45309] text-white shadow-md font-bold';
            btn.className = 'add-lead-tab flex-1 py-2 text-center text-[13px] rounded-lg transition-all ' + activeStyle;
        } else {
            btn.className = 'add-lead-tab flex-1 py-2 text-center text-[13px] font-semibold rounded-lg transition-all text-slate-500 hover:text-slate-800 bg-transparent';
        }
    });
    document.querySelectorAll('.add-lead-panel').forEach((panel, i) => {
        panel.classList.toggle('hidden', i !== tabIndex);
    });
    const backBtn = document.getElementById('add-lead-back-btn');
    const nextBtn = document.getElementById('add-lead-next-btn');

function updateSelectedUserStatus() {
    const sel = document.getElementById('lead-user');
    const badge = document.getElementById('selected-user-status');
    if (!sel || !badge) return;
    
    const selectedVal = sel.value;
    if (!selectedVal) {
        badge.classList.add('hidden');
        return;
    }
    
    let userStatus = 'Offline';
    try {
        const stored = localStorage.getItem('leadUsersData');
        if (stored) {
            const list = JSON.parse(stored);
            if (Array.isArray(list)) {
                const user = list.find(u => u.fullname === selectedVal);
                if (user) {
                    userStatus = user.status || 'Offline';
                }
            }
        }
    } catch(e) {}
    
    badge.textContent = userStatus;
    badge.classList.remove('hidden');
    if (userStatus === 'Online') {
        badge.className = 'text-[10px] font-bold px-2 py-0.5 rounded-full bg-emerald-100 text-emerald-800 border border-emerald-250 ml-2';
    } else {
        badge.className = 'text-[10px] font-bold px-2 py-0.5 rounded-full bg-slate-100 text-slate-600 border border-slate-200 ml-2';
    }
}

function handleLeadSubmit(event) {
    event.preventDefault();
    const originalName = document.getElementById('edit-lead-original-name').value;
    const name = document.getElementById('lead-name').value.trim();
    if (name.length < 3) {
        document.getElementById('lead-name-error').textContent = 'Atleast 3 letter or number is mandatory.Space Allow with alphanumberic string';
        document.getElementById('lead-name-error').classList.remove('hidden');
        switchAddLeadTab(0);
        return;
    }
    document.getElementById('lead-name-error').classList.add('hidden');

    // Duplicate Data Validation Check
    const duplicateNameLead = leadsData.find(l => {
        if (originalName && l.name === originalName) return false;
        return l.name && l.name.trim().toLowerCase() === name.toLowerCase();
    });
    if (duplicateNameLead) {
        document.getElementById('lead-name-error').textContent = 'A lead with this Name already exists.';
        document.getElementById('lead-name-error').classList.remove('hidden');
        switchAddLeadTab(0);
        document.getElementById('lead-name').focus();
        return;
    }

    const mobileInput = document.getElementById('lead-mobile').value.trim();
    const duplicateMobileLead = (mobileInput && mobileInput !== '-') ? leadsData.find(l => {
        if (originalName && l.name === originalName) return false;
        return l.mobile && l.mobile !== '-' && l.mobile.trim() === mobileInput;
    }) : null;
    if (duplicateMobileLead) {
        alert('Error: A lead with this Mobile number already exists (' + duplicateMobileLead.name + ').');
        switchAddLeadTab(0);
        document.getElementById('lead-mobile').focus();
        return;
    }
    const email = document.getElementById('lead-email').value.trim() || '-';
    const countryCode = document.getElementById('lead-country-code').value;
    const mobile = document.getElementById('lead-mobile').value.trim() || '-';
    const stage = document.getElementById('lead-stage').value || 'LEAD';
    const source = document.getElementById('lead-source').value || '-';
    const user = document.getElementById('lead-user').value || '-';
    const segment = document.getElementById('lead-segment').value || '';
    const priority = document.getElementById('lead-priority').value || 'Low';
    const altMobile1 = document.getElementById('lead-alt-mobile-1').value.trim();
    const altMobile2 = document.getElementById('lead-alt-mobile-2').value.trim();
    const altMobile3 = document.getElementById('lead-alt-mobile-3').value.trim();
    const age = document.getElementById('lead-age').value.trim();
    const genderEl = document.querySelector('input[name="lead-gender"]:checked');
    const gender = genderEl ? genderEl.value : 'Male';
    const address = document.getElementById('lead-address').value.trim();
    const state = document.getElementById('lead-state').value;
    const city = document.getElementById('lead-city').value;
    const country = document.getElementById('lead-country').value.trim();
    const pincode = document.getElementById('lead-pincode').value.trim();
    const company = document.getElementById('lead-company').value.trim();
    const designation = document.getElementById('lead-designation').value.trim();
    const website = document.getElementById('lead-website').value.trim();
    const tags = document.getElementById('lead-tags').value;
    const followupDate = document.getElementById('lead-followup').value;
    const opportunity = document.getElementById('lead-opportunity').value.trim();
    const customFields = [];
    document.querySelectorAll('.custom-field-row').forEach(row => {
        const fn = row.querySelector('.custom-field-name');
        const fv = row.querySelector('.custom-field-value');
        const fname = fn ? fn.value.trim() : '';
        const fvalue = fv ? fv.value.trim() : '';
        if (fname || fvalue) customFields.push({name: fname, value: fvalue});
    });
    const amount = opportunity || '-';
    const now = new Date().toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric', hour:'2-digit', minute:'2-digit' });

    if (originalName) {
        const lead = leadsData.find(l => l.name === originalName);
        if (lead) {
            Object.assign(lead, {
                name, email, mobile, countryCode, stage, source, user, amount,
                status: stage, priority, segment,
                altMobile1, altMobile2, altMobile3, age, gender,
                address, state, city, country, pincode,
                company, designation, website, tags, opportunity, followupDate,
                customFields, modifiedDate: now
            });
        }
    } else {
        leadsData.push({
            name, stage, email, mobile, countryCode,
            status: stage, source, amount, user, priority, segment,
            altMobile1, altMobile2, altMobile3, age, gender,
            address, state, city, country, pincode,
            company, designation, website, tags, opportunity, followupDate,
            customFields, completedDate: '',
            creationDate: now, modifiedDate: now,
            date: Date.now(), callsCount: 0, smsCount: 0, whatsappCount: 0
        });
    }
    saveLeadsData();
    closeLeadModal();
    renderLeads();
}

            try {
                let list = [];

                        name: val.toUpperCase(),

                    });
                    localStorage.setItem('leadSourcesData', JSON.stringify(list));

        } else if (selectId === 'lead-stage') {
            try {
                let list = [];
                const stored = localStorage.getItem('leadStagesData');

                    });

function toggleDropdown(event, id) {
    event.stopPropagation();
    document.querySelectorAll('.dropdown-menu').forEach(m => { if (m.id !== id) m.classList.add('hidden'); });
    
    const dropdown = document.getElementById(id);
    const isHidden = dropdown.classList.contains('hidden');
    
    if (isHidden) {
        dropdown.classList.remove('hidden');
        
        const button = event.currentTarget;
        const rect = button.getBoundingClientRect();
        
        dropdown.style.position = 'fixed';
        dropdown.style.marginTop = '0px';
        dropdown.style.top = `${rect.bottom + 6}px`;
        
        if (id === 'customize-dropdown' || id === 'utilities-dropdown' || id === 'admin-menu') {
            dropdown.style.right = `${window.innerWidth - rect.right}px`;
            dropdown.style.left = 'auto';
        } else {
            dropdown.style.left = `${rect.left}px`;
            dropdown.style.right = 'auto';
        }
    } else {
        dropdown.classList.add('hidden');
    }
}

// Live clock init function
function initClock() {
    const clockEl = document.getElementById('live-clock');
    if (!clockEl) return;
    const updateTime = () => {
        const now = new Date();
        const dd = String(now.getDate()).padStart(2, '0');
        const mm = String(now.getMonth() + 1).padStart(2, '0');
        const yyyy = now.getFullYear();
        const hh = String(now.getHours()).padStart(2, '0');
        const min = String(now.getMinutes()).padStart(2, '0');
        const ss = String(now.getSeconds()).padStart(2, '0');
        clockEl.textContent = `${dd}-${mm}-${yyyy} ${hh}:${min}:${ss}`;
    };
    updateTime();
    setInterval(updateTime, 1000);
}

// Fullscreen toggle function
function toggleFullscreen(button) {
        if (!document.fullscreenElement) {
            document.documentElement.requestFullscreen().then(() => {
                localStorage.setItem('fullscreen', 'enabled');
            }).catch(err => {
                console.log(`Error attempting to enable full-screen mode: ${err.message}`);
            });
        } else {
            document.exitFullscreen().then(() => {
                localStorage.setItem('fullscreen', 'disabled');
            }).catch(e => {});
        }
    }

    // Auto-restore fullscreen on first interaction if it was enabled
    if (!window.hasFullscreenRestorer) {
        window.hasFullscreenRestorer = true;
        const restoreFullscreen = () => {
            if (localStorage.getItem('fullscreen') === 'enabled' && !document.fullscreenElement) {
                document.documentElement.requestFullscreen().catch(e => {});
            }
            document.removeEventListener('click', restoreFullscreen);
            document.removeEventListener('keydown', restoreFullscreen);
        };
        document.addEventListener('click', restoreFullscreen);
        document.addEventListener('keydown', restoreFullscreen);
    }

    function handleHeaderChangePassword(event) {
        event.preventDefault();
        const form = event.target;
        const oldPass = form.querySelector('.header-old-pass').value;
        const newPass = form.querySelector('.header-new-pass').value;
        const confirmPass = form.querySelector('.header-confirm-pass').value;

        const session = JSON.parse(localStorage.getItem('crmActiveSession') || 'null');
        if (!session) {
            showToast('No active session found.', 'error');
            return;
        }

        let users = [];
        try { users = JSON.parse(localStorage.getItem('crmUsers') || '[]'); } catch(e) {}
        
        const userIdx = users.findIndex(u => u.username.toLowerCase() === session.username.toLowerCase());
        if (userIdx === -1) {
            showToast('User not found.', 'error');
            return;
        }

        if (oldPass !== users[userIdx].password) {
            showToast('Incorrect old password!', 'error');
            return;
        }

        if (newPass.length < 6) {
            showToast('New password must be at least 6 characters!', 'error');
            return;
        }

        if (newPass !== confirmPass) {
            showToast('New password and confirm password do not match!', 'error');
            return;
        }

        if (newPass === oldPass) {
            showToast('New password cannot be the same as old password!', 'error');
            return;
        }

        users[userIdx].password = newPass;

    function formatHeaderDateTime(date) {

        const yyyy = date.getFullYear();
        const hh = String(date.getHours()).padStart(2, '0');

        return `${dd}-${mm} ${hh}:${min}`;
    }

function parseCSVLine(line) {
    let result = [];
    let current = '';
    let inQuotes = false;
    for (let i = 0; i < line.length; i++) {
        let char = line[i];
        if (char === '"') {
            inQuotes = !inQuotes;
        } else if (char === ',' && !inQuotes) {
            result.push(current.trim());
            current = '';
        } else {
            current += char;
        }
    }
    result.push(current.trim());
    return result;
}

function parseCSVContent(text) {
    let lines = text.split('\n');
    let rows = [];
    for (let i = 0; i < lines.length; i++) {
        let line = lines[i].replace(/\r$/, '').trim();
        if (line) {
            rows.push(parseCSVLine(line));
        }
    }
    return rows;
}

function handleCSVFileSelection(event) {
    const file = event.target.files[0];
    if (!file) return;
    currentCSVFileName = file.name;

    
    if (!file.name.toLowerCase().endsWith('.csv')) {
        showToast('Please select a valid CSV file.', 'error');
        return;
    }
    
    if (file.size > 5 * 1024 * 1024) {
        showToast('File size exceeds the 5 MB limit.', 'error');
        return;
    }
    
    const reader = new FileReader();
    reader.onload = function(e) {
        const text = e.target.result;
        const rows = parseCSVContent(text);
        
        if (rows.length === 0) {
            showToast('The CSV file is empty.', 'error');
            return;
        }
        
        parsedCSVHeaders = rows[0];
        parsedCSVRows = rows.slice(1);
        
        document.getElementById('csv-upload-status').innerHTML = `<span class="text-teal-600 font-extrabold flex items-center gap-1"><span class="material-symbols-outlined text-[18px]">check_circle</span> ${file.name}</span>`;
        document.getElementById('csv-total-rows').value = parsedCSVRows.length;
        document.getElementById('csv-details-card').classList.remove('hidden');
        document.getElementById('import-next-btn').disabled = false;
        
        showToast('CSV loaded successfully. Select defaults below.');
    };
    reader.readAsText(file);
}

function executeLeadsImport() {
    const selects = document.querySelectorAll('.csv-mapper-select');
    let mapping = {};
    selects.forEach(sel => {
        const val = sel.value;
        const idx = parseInt(sel.getAttribute('data-header-index'));
        if (val) {
            mapping[val] = idx;
        }
    });
    
    if (!mapping['name'] && !mapping['mobile'] && !mapping['email']) {
        showToast('Please map at least Name, Mobile, or Email to import.', 'error');
        return;
    }
    
    const defaultSegment = document.getElementById('csv-default-segment').value;
    const defaultUserGroup = document.getElementById('csv-default-usergroup').value;
    const defaultUser = document.getElementById('csv-default-user').value;
    const defaultStage = document.getElementById('csv-default-stage').value;
    const defaultSource = document.getElementById('csv-default-source').value;
    
    let importCount = 0;
    const batchSeenNames = new Set();
    const batchSeenMobiles = new Set();
    
    parsedCSVRows.forEach(row => {
        const nameVal = mapping['name'] !== undefined ? row[mapping['name']] : '';
        const mobileVal = mapping['mobile'] !== undefined ? row[mapping['mobile']] : '';
        const emailVal = mapping['email'] !== undefined ? row[mapping['email']] : '';
        const companyVal = mapping['company'] !== undefined ? row[mapping['company']] : '';
        const segmentVal = mapping['segment'] !== undefined ? row[mapping['segment']] : '';
        const followupVal = mapping['followup'] !== undefined ? row[mapping['followup']] : '';
        const addressVal = mapping['address'] !== undefined ? row[mapping['address']] : '';
        const cityVal = mapping['city'] !== undefined ? row[mapping['city']] : '';
        const stateVal = mapping['state'] !== undefined ? row[mapping['state']] : '';
        const pincodeVal = mapping['pincode'] !== undefined ? row[mapping['pincode']] : '';
        const userVal = mapping['user'] !== undefined ? row[mapping['user']] : '';
        const sourceVal = mapping['source'] !== undefined ? row[mapping['source']] : '';
        const statusVal = mapping['status'] !== undefined ? row[mapping['status']] : '';
        
        if (!nameVal && !mobileVal && !emailVal) return;
        
        const cleanName = (nameVal || '').trim();
        const cleanMobile = (mobileVal || '').trim();
        
        // Check for duplicates in existing database or in this current import batch
        const nameExists = cleanName && (
            leadsData.some(l => l.name && l.name.trim().toLowerCase() === cleanName.toLowerCase()) ||
            batchSeenNames.has(cleanName.toLowerCase())
        );
        const mobileExists = cleanMobile && cleanMobile !== '-' && (
            leadsData.some(l => l.mobile && l.mobile !== '-' && l.mobile.trim() === cleanMobile) ||
            batchSeenMobiles.has(cleanMobile)
        );
        
        if (nameExists || mobileExists) {
            // Skip duplicate
            return;
        }
        
        // Mark as seen
        if (cleanName) batchSeenNames.add(cleanName.toLowerCase());
        if (cleanMobile && cleanMobile !== '-') batchSeenMobiles.add(cleanMobile);
        
        const now = new Date().toLocaleDateString('en-GB') + " " + new Date().toLocaleTimeString('en-US', {hour: '2-digit', minute:'2-digit'});
        
        const newLead = {
            id: 'LD-' + Math.floor(100000 + Math.random() * 900000),
            name: cleanName || 'CSV Lead',
            mobile: cleanMobile || '-',
            email: emailVal ? emailVal.trim() : '-',
            company: companyVal ? companyVal.trim() : (defaultSegment || '-'),
            segment: segmentVal ? segmentVal.trim() : (defaultSegment || 'Enterprise'),
            followupDate: followupVal ? followupVal.trim() : '',
            address: addressVal ? addressVal.trim() : '-',
            city: cityVal ? cityVal.trim() : '-',
            state: stateVal ? stateVal.trim() : '-',
            pincode: pincodeVal ? pincodeVal.trim() : '-',
            user: userVal ? userVal.trim() : (defaultUser || '-'),
            userGroup: defaultUserGroup || 'Sales Team',
            source: sourceVal ? sourceVal.trim() : (defaultSource || 'Direct Sales'),
            stage: statusVal ? statusVal.trim().toUpperCase() : (defaultStage || 'LEAD'),
            amount: '0.00',
            score: 0,
            gender: 'Male',
            priority: 'Medium',
            customFields: {},
            creationDate: now,
            modifiedDate: now,
            date: Date.now(),
            callsCount: 0,
            smsCount: 0,
            whatsappCount: 0
        };
        
        leadsData.push(newLead);
        importCount++;
    });
    
    const totalRecords = parsedCSVRows.length;
    const createdRecords = importCount;
    const invalidRecords = totalRecords - createdRecords;
    
    saveImportHistoryLog(currentCSVFileName || 'imported_leads.csv', totalRecords, createdRecords, invalidRecords, 'CSV');
    
    if (importCount > 0) {
        saveLeadsData();
        renderLeads();
        showToast(`Successfully imported ${importCount} leads!`, 'success');
    } else {
        showToast('No leads were imported.', 'error');
    }
    
    closeImportLeadsModal();
}

        }
        
        const newLog = {
            id: 'IMP-' + Math.floor(100000 + Math.random() * 900000),
            fileName: fileName,
            totalRecords: total,
            createdRecords: created,
            invalidRecords: invalid,
            date: nowStr,
            status: status,
            uploadedBy: username,
            importType: type
        };
        
        history.unshift(newLog);
        localStorage.setItem('crmImportHistory', JSON.stringify(history));
    } catch(e) {
        console.error("Error saving import log:", e);
    }
}

function renderImportHistory() {
    const historyData = getImportHistoryData();
    
    let filtered = historyData;
    if (historySearchQuery) {
        const query = historySearchQuery.toLowerCase();
        filtered = historyData.filter(log => 
            log.fileName.toLowerCase().includes(query) ||
            log.status.toLowerCase().includes(query) ||
            log.uploadedBy.toLowerCase().includes(query) ||
            log.importType.toLowerCase().includes(query)
        );
    }
    
    const total = filtered.length;
    const totalPages = Math.max(1, Math.ceil(total / historyPageSize));
    
    if (historyCurrentPage > totalPages) {
        historyCurrentPage = totalPages;
    }
    
    const startIdx = (historyCurrentPage - 1) * historyPageSize;
    const endIdx = Math.min(startIdx + historyPageSize, total);
    
    const paginated = filtered.slice(startIdx, endIdx);
    
    const tbody = document.getElementById('import-history-table-body');
    const emptyState = document.getElementById('history-empty-state');
    
    tbody.innerHTML = '';
    
    if (paginated.length === 0) {
        emptyState.classList.remove('hidden');
    } else {
        emptyState.classList.add('hidden');
        paginated.forEach((log, index) => {
            const tr = document.createElement('tr');
            tr.className = 'hover:bg-transparent transition-colors border-b border-slate-100';
            
            let statusBadge = '';
            if (log.status === 'Completed') {
                statusBadge = '<span class="px-2.5 py-1 bg-emerald-50 text-emerald-700 border border-emerald-200 rounded-full text-[10px] font-bold">Completed</span>';
            } else if (log.status === 'Completed with errors') {
                statusBadge = '<span class="px-2.5 py-1 bg-amber-50 text-amber-700 border border-amber-200 rounded-full text-[10px] font-bold">Completed with errors</span>';
            } else {
                statusBadge = '<span class="px-2.5 py-1 bg-rose-50 text-rose-700 border border-rose-200 rounded-full text-[10px] font-bold">Failed</span>';
            }
            
            const srNo = startIdx + index + 1;
            
            tr.innerHTML = `
                <td class="py-3 px-4 text-center font-bold text-slate-500">${srNo}</td>
                <td class="py-3 px-4 font-bold text-slate-800 flex items-center gap-2">
                    <span class="material-symbols-outlined text-[18px] text-indigo-500">description</span>
                    <span>${log.fileName}</span>
                </td>
                <td class="py-3 px-4 text-right font-semibold text-slate-600">${log.totalRecords}</td>
                <td class="py-3 px-4 text-right font-bold text-emerald-600">${log.createdRecords}</td>
                <td class="py-3 px-4 text-right font-bold ${log.invalidRecords > 0 ? 'text-rose-500' : 'text-slate-400'}">${log.invalidRecords}</td>
                <td class="py-3 px-4 text-slate-500 font-semibold">${log.date}</td>
                <td class="py-3 px-4 text-center">${statusBadge}</td>
                <td class="py-3 px-4 text-slate-700 font-bold">${log.uploadedBy}</td>
                <td class="py-3 px-4 text-center">
                    <span class="px-2 py-0.5 bg-slate-100 text-slate-600 border border-slate-200 rounded text-[10px] font-extrabold uppercase">${log.importType}</span>
                </td>
                <td class="py-3 px-4 text-center">
                    <button onclick="deleteHistoryLog('${log.id}', event)" class="p-1 hover:bg-rose-50 text-rose-500 hover:text-rose-700 rounded transition-colors border-none bg-transparent cursor-pointer flex items-center justify-center mx-auto" title="Delete log">
                        <span class="material-symbols-outlined text-[16px]">delete</span>
                    </button>
                </td>
            `;
            tbody.appendChild(tr);
        });
    }
    
    document.getElementById('history-pagination-info').textContent = total > 0 
        ? `Showing ${startIdx + 1}-${endIdx} of ${total} entries` 
        : 'Showing 0-0 of 0 entries';
        
    document.getElementById('history-current-page').textContent = `Page ${historyCurrentPage} of ${totalPages}`;
    
    document.getElementById('history-prev-btn').disabled = (historyCurrentPage === 1);
    document.getElementById('history-next-btn').disabled = (historyCurrentPage === totalPages);
}

function filterImportHistory() {
    historySearchQuery = document.getElementById('import-history-search').value;
    historyCurrentPage = 1;
    renderImportHistory();
}

function historyPageChange(direction) {
    historyCurrentPage += direction;
    renderImportHistory();
}

function toggleHistoryPageSizeDropdown(event) {
    event.stopPropagation();
    const dropdown = document.getElementById('history-pages-dropdown');
    if (dropdown) dropdown.classList.toggle('hidden');
}

function changeHistoryPageSize(size) {
    historyPageSize = size;
    const label = document.getElementById('history-pages-label');
    if (label) label.textContent = `${size} / Page`;
    const dropdown = document.getElementById('history-pages-dropdown');
    if (dropdown) dropdown.classList.add('hidden');
    historyCurrentPage = 1;
    renderImportHistory();
}

function triggerImportFromHistory() {
    closeImportHistoryModal();
    openImportLeadsModal();
}

function triggerRefreshHistory() {
    renderImportHistory();
    showToast("Import history logs refreshed successfully!", "success");
}

function deleteHistoryLog(id, event) {
    if (event) event.stopPropagation();
    if (!confirm("Are you sure you want to delete this import log?")) return;
    
    let history = [];
    try {
        const stored = localStorage.getItem('crmImportHistory');
        if (stored) {
            history = JSON.parse(stored);
        }
    } catch(e) {}
    
    if (Array.isArray(history)) {
        history = history.filter(item => item.id !== id);
        localStorage.setItem('crmImportHistory', JSON.stringify(history));
    }
    
    renderImportHistory();
    showToast("Import log deleted successfully!", "success");
}

function clearAllHistory(event) {
    if (event) event.stopPropagation();
    if (!confirm("Are you sure you want to delete ALL import history logs? This cannot be undone.")) return;
    
    localStorage.setItem('crmImportHistory', JSON.stringify([]));
    
    renderImportHistory();
    showToast("All import logs cleared!", "success");
}
